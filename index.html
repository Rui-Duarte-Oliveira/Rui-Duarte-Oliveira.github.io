void AAIController::EvaluateTarget() {
    float BestScore = 0.f;
    AActor* BestTarget = nullptr;

    for (auto& Stimulus : PerceptionComp->GetPerceivedActors()) {
        float Score = CalcThreatScore(Stimulus);
        Score *= GetDistanceWeight(Stimulus);
        
        if (Score > BestScore) {
            BestScore = Score;
            BestTarget = Stimulus;
        }
    }
    
    SetCurrentTarget(BestTarget);
}
